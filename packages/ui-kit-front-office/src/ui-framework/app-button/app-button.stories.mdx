import { Meta, Story, Canvas, Source, ArgsTable } from '@storybook/addon-docs';
import AppButton from './app-button.vue';
import { APP_BUTTON_TYPES, APP_BUTTON_SIZES } from './constants';
const argTypes = {
  text: { control: 'text' },
  disabled: { control: 'boolean' },
  outlined: { control: 'boolean' },
  isLoading: { control: 'boolean' },
  onlyIcon: { control: 'boolean' },
  type: { control: { type: 'radio', options: Object.values(APP_BUTTON_TYPES) } },
  size: { control: { type: 'radio', options: Object.values(APP_BUTTON_SIZES) } },
  nativeType: { control: { type: 'radio', options: ['button', 'submit', 'reset'] } },
};

<Meta
  title="ui-framework/AppButton"
  component={AppButton}
  argTypes={argTypes}
/>

export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { AppButton },
  template: `
      <app-button
        :type="type"
        :icon="icon"
        :iconRight="iconRight"
        :onlyIcon="onlyIcon"
        :disabled="disabled"
        :isLoading="isLoading"
        :outlined="outlined"
        :size="size"
        :fullWidth="fullWidth"
        :rounded="rounded"
      >
      {{ text }}
      </app-button>`,
});
const DEFAULT_PROPS = {
  nativeType: 'button',
  size: APP_BUTTON_SIZES.big,
  type: APP_BUTTON_TYPES.primary,
  text: 'App button',
};
