import { Meta, Story, Canvas, Source, ArgsTable } from '@storybook/addon-docs';
import AppButton from '@/components/ui-framework/app-button.vue'
const argTypes = {
    text: { control: { type: 'text', label: 'Button text' } },
};

<Meta title="ui-framework/AppButton" component={AppButton} argTypes={argTypes}/>

export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { AppButton },
  template: `<app-button
    :icon='icon'
    :iconRight='iconRight'
  >
    {{ text }}
  </app-button>`,
});


## AppButton

Example:
<Source code={`<app-button>Some text</app-button>`} ></Source>

### Default
<Canvas>
  <Story name="Default view" args={{ text: 'Some text' }}>
    { Template.bind({}) }
  </Story>
</Canvas>

### With icon
To set an icon you need just pass prop **icon** with icon class name.
<Canvas>
  <Story name="Icon left" args={{ text: 'Some text', icon: 'icon-user-profile' }}>
    { Template.bind({}) }
  </Story>
</Canvas>

### With right side icon
To place an icon to right side, just pass **iconRight** prop.
<Canvas>
  <Story name="Icon right" args={{ text: 'Some text', icon: 'icon-user-profile', iconRight: true }}>
    { Template.bind({}) }
  </Story>
</Canvas>
